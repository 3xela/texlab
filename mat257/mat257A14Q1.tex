\noindent Q1: Prove that $d(fg) = g\cdot df + f\cdot dg$. We compute that 
\begin{align*}
d(fg) & = \sum_{i=1}^n \frac{\partial(f\cdot g)}{\partial x_i} dx_i
\\ & = \sum_{i=1}^n \Big[g\cdot \frac{\partial f}{\partial x_i}+ f\cdot \frac{\partial g}{\partial x_i} \Big]dx_i
\\ & = g\cdot \sum_{i=1}^n \frac{\partial f}{\partial x_i} dx_i + f\cdot \sum_{i=1}^n \frac{\partial g}{\partial x_i} dx_i
\\ & = g\cdot df + f\cdot dg
\end{align*}
