\noindent Q2a: Letting $s(x) = \sum_{k=0}^n a_k x^k$, with $a_n=1$, we see that 
\begin{align*}
    s(\alpha) & = [x^n]_{s(x)} + a_{n-1}[x^{n-1}]_{s(x)}  + \dots + a_0
    \\ & = [x^n]_{s(x)} + [a_{n-1}x^{n-1}]_{s(x)} + \dots +[a_0]_{s(x)}
    \\ & = [x^n + a_{n-1}x^{n-1} + \dots a_0]_{s(x)}
    \\ & = [s(x)]_{s(x)}
    \\ & = [0]_{s(x)}
\end{align*}
